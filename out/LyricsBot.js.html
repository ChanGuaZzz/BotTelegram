const ClientSecret = process.env.ClientSecret;

/**
 * Obtiene el token de acceso para la API de Spotify utilizando el flujo de credenciales de cliente.
 * @async
 * @function
 * @returns {Promise<string>} Token de acceso para la API de Spotify.
 */
async function getAccessToken() {
  const response = await axios.post('https://accounts.spotify.com/api/token', null, {
    params: {
      grant_type: 'client_credentials',
    },
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded',
    },
    auth: {
      username: ClientID,
      password: ClientSecret,
    },
  });

  return response.data.access_token;
}

/**
 * Busca artistas seg√∫n el g√©nero proporcionado.
 * @async
 * @function
 * @param {string} genre - G√©nero musical para buscar artistas.
 * @returns {Promise<Array>} Lista de artistas que coinciden con el g√©nero.
 */
async function searchArtistsByGenre(genre) {
  const accessToken = await getAccessToken();

  const response = await axios.get('https://api.spotify.com/v1/search', {
    params: {
      q: `genre:${genre}`,
      type: 'artist',
    },
    headers: {
      Authorization: `Bearer ${accessToken}`,
    },
  });

  return response.data.artists.items;
}

/**
 * Obtiene la URL de la letra de una canci√≥n a trav√©s de la API de Genius.
 * @async
 * @function
 * @param {string} SongArtist - Nombre de la canci√≥n y el artista.
 * @returns {Promise<?string>} URL de la letra de la canci√≥n.
 */
async function getLyrics(SongArtist) {
  try {
    const response = await axios.get(
      `https://api.genius.com/search?q=${encodeURIComponent(`${SongArtist}`)}`,
      {
        headers: {
          Authorization: `Bearer ${apiKey}`,
        },
      }
    );

    const url = response.data.response.hits[0].result.url;
    return url;
  } catch (error) {
    console.error("Error al obtener la letra:", error);
    return null;
  }
}
